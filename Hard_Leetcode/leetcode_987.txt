/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void reqAnswer(TreeNode* root, map<int, map<int, multiset<int>>> &nodes) {
        if (root==NULL) return;
        queue<pair<TreeNode*, pair<int, int>>> q; 
        q.push({root, {0, 0}});
        while (!q.empty()) { 
            int len=q.size();
            for (int i=0; i<len; i++) {
                auto temp = q.front();
                q.pop();
                TreeNode* node = temp.first; // the current node whose left & right is to be checked
                int vertical = temp.second.first; // initialise the vertical with prev stored value
                int level = temp.second.second; // initialise the level with prev stored value
                nodes[vertical][level].insert(node->val);
                if (node->left!=NULL) {
                    q.push({node->left, {vertical - 1, level + 1}});
                }
                if (node->right!=NULL) {
                    q.push({node->right, {vertical + 1, level + 1}});
                }
            }
        }
    }

    vector<vector<int>> verticalTraversal(TreeNode* root) {
        vector<vector<int>> ans;
        map<int, map<int, multiset<int>>> nodes;
        reqAnswer(root, nodes);
        for (auto p : nodes) { // traverse the map
            vector<int> col;
            for (auto q : p.second) { // traverse the map inside the map
                col.insert(col.end(), q.second.begin(), q.second.end()); // access the multiset, pushing the multiset in the vector
            }
            ans.push_back(col);
        }
        return ans;
    }
};

class Solution {
public:
    void dfs(string word, string &beginWord, vector<string> &seq, unordered_map <string,int> &mpp, vector<vector<string>> &ans) {
        // If the sequence is complete
        if(word == beginWord) {
            reverse(seq.begin(), seq.end());
            ans.push_back(seq);
            
            // Reverse again for other possible sequences
            reverse(seq.begin(), seq.end());
            return;
        }
        
        // Steps to reach the word
        int val = mpp[word];
        
        for(int i=0; i < word.length(); i++) { // Try all possible transformations
            char original = word[i];
            
            for(char ch = 'a'; ch <= 'z'; ch++) {
                word[i] = ch;
                
                // If a possible transformations is found
                if(mpp.find(word) != mpp.end() && 
                   mpp[word] + 1 == val) {
                       
                    // Update the sequence
                    seq.push_back(word);
                    
                    // Make recursive DFS call
                    dfs(word, beginWord, seq, mpp, ans);
                    
                    // Pop back for backtracking
                    seq.pop_back();
                }
            }
            word[i] = original;
        }
    }

    vector<vector<string>> findLadders(string beginWord, string endWord, vector<string>& wordList) {
        int len = beginWord.length();
        
        // Add all the words to a hashset
        unordered_set <string> st(wordList.begin(), wordList.end());
                                  
        // Hash map to store the minimum steps needed
        unordered_map <string,int> mpp;
        
        queue <string> q;
        q.push(beginWord);
        st.erase(beginWord);
        int steps = 1;
        mpp[beginWord] = steps;
        
        while(!q.empty()) {
            string word = q.front();
            steps = mpp[word];
            q.pop();

            // Check for every possible transformation
            for(int i = 0; i < len; i++) {
                char original = word[i];
                
                for(char ch = 'a'; ch <= 'z'; ch++) {
                    word[i] = ch;
                    
                    // If a possible transformation is found
                    if(st.find(word) != st.end()) {
                        
                        // Store it in map
                        mpp[word] = steps+1;
                        
                        // Push in the queue
                        q.push(word);
                        
                        // Erase word from list
                        st.erase(word);
                    }
                }
                word[i] = original;
            }
        }

        
        if(mpp.find(endWord) == mpp.end()) return {};

        vector<vector<string>> ans;
        
        // To store the possible sequence of transformations
        vector <string> seq = {endWord};
        
        // Backtracking step
        dfs(endWord, beginWord, seq, mpp, ans);
        
        return ans;
    }
};

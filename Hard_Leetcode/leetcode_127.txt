class Solution {
public:
    int ladderLength(string beginWord, string endWord, vector<string>& wordList) {
        queue<pair<string, int>> q;
        q.push({beginWord, 1});
        unordered_set<string> st(wordList.begin(), wordList.end());
        
        if(st.find(endWord) == st.end()) return 0; // If target word is not in word list, return 0 

        st.erase(beginWord);  //Erase the starting word from set (if it exists)
        
        while (!q.empty()){
            string word = q.front().first;
            int steps = q.front().second;
            q.pop();

            if (word == endWord) // Return steps if target word is achieved
                return steps;
    
            for (int i=0; i < word.size(); i++) { // Iterate on every character 
                char original = word[i];
                
                for (char ch = 'a'; ch <= 'z'; ch++) {
                    word[i] = ch;

                    if (st.find(word) != st.end()) {
                        st.erase(word);
                        q.push({word, steps + 1});
                    }
                }
                
                // Update the original letter back
                word[i] = original;
            }
        }
        return 0;
    }
};

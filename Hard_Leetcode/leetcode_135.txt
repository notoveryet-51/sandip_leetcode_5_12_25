class Solution {
public:
    int candy(vector<int>& ratings) {
        int n=ratings.size();
        vector<int> candies(n, 1); // start with assigning 1 candy to all
        // Left to right
        for (int i=1; i<n; i++) {
            if (ratings[i]>ratings[i-1]) candies[i]=candies[i-1]+1;
        }
        // Right to left
        for (int i=n-2; i>=0; i--) {
            if (ratings[i]>ratings[i+1]) candies[i]=max(candies[i], candies[i+1]+1);
        }
        int ans=0;
        for (int x: candies) ans+=x;
        return ans;
    }
};

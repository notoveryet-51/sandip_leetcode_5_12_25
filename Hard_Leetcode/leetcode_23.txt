/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        vector<int> total;
        for (auto it: lists) {
            ListNode *temp=it;
            while (temp) {
                total.push_back(temp->val);
                temp=temp->next;
            }
        }
        if (total.empty()) return NULL;
        sort(total.begin(), total.end());
        ListNode *head=new ListNode(total[0]);
        ListNode *tail=head;
        for (int i=1; i<total.size(); i++) {
            ListNode *newNode= new ListNode(total[i]);
            tail->next=newNode;
            tail=tail->next;
        }
        return head;
    }
};

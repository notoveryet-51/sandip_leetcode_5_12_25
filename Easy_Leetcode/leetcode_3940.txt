class Solution {
public:
    vector<int> limitOccurrences(vector<int>& nums, int k) {
        vector<int> result;
        int n=nums.size();
        unordered_map<int,int>mpp;
        for (int i=0; i<n; i++) {
            if (mpp[nums[i]]<k) {
                result.push_back(nums[i]);
                mpp[nums[i]]++;
            }
        }
        return result;
    }
};

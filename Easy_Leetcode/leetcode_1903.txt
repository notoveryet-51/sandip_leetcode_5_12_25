class Solution {
public:
    string largestOddNumber(string num) {
        string s="";
        if (((num.back() - '0') & 1) != 0)   return num;
        int index=-1;
        for (int i=0; i<num.size(); i++) {
            if (((num[i] - '0') & 1) != 0)  index=i;
        }
        if (index==-1)   return s;
        else return num.substr(0,index+1); 
    }
};

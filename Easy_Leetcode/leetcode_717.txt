class Solution {
public:
    bool isOneBitCharacter(vector<int>& bits) {
        int count=0,n=bits.size();
        for (int i=n-2; i>=0; i--)  {
            if (bits[i]==1) count++;
            else break;
        }
        if ((count&1)==0)   return true;
        else return false;
    }
};

class Solution {
public:
    int findGCD(vector<int>& nums) {
        int maxi=INT_MIN;
        int mini=INT_MAX;
        for (int i=0; i<nums.size(); i++) {
            if (nums[i]>maxi)   maxi=nums[i];
            if (nums[i]<mini)   mini=nums[i];
        }
        for (int i=mini; i>0; i--) {
            if (maxi%i==0 && mini%i==0) return i;
        }
        return 0;
    }
};

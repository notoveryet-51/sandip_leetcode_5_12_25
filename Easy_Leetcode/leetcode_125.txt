class Solution {
public:
    bool isPalindrome(string s) {
        for (int i=0; i<s.size(); i++)  s[i]=tolower(s[i]);
        string empty="";
        for (int i=0; i<s.size(); i++) if ((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= '0' && s[i] <= '9')) empty+=s[i];
        string compare=empty;
        reverse(empty.begin(),empty.end());
        if (compare==empty) return true;
        return false;
    }
};

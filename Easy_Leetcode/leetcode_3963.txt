class Solution {
public:
    vector<string> createGrid(int m, int n) {
        vector<string>res;
        for (int j=0; j<m-1; j++) {
            string ini=".";
            for (int i=0; i<n-1; i++) ini+="#";
            res.push_back(ini);
        }
        string fin="";
        for (int i=0; i<n; i++) fin+=".";
        res.push_back(fin);
        return res;
    }
};

Leetcode 1614

class Solution {
public:
    int maxDepth(string s) {
        int currentDepth = 0;
        int maxi = 0;
        for (auto ch : s) {
            if (ch == '(') {
                currentDepth++;
                maxi = max(maxi, currentDepth);
            }
            else if (ch == ')') {
                currentDepth--;
            }
        }
        return maxi;
    }
};

class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        map<int,int> mp;
        for (int i = 0; i < nums.size(); i++) {
            auto it = mp.find(nums[i]);
            if (it != mp.end()) {
                int check = abs(i - it->second);
                if (check <= k) return true;
            }
            mp[nums[i]] = i;  
        }
        return false;
    }
};

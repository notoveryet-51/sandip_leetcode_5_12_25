class Solution {
public:
    int residuePrefixes(string s) {
        set<char>st;
        int n=s.size();
        int count=1;
        st.insert(s[0]);
        for (int j=1; j<n; j++) {
            st.insert(s[j]);
            if ((j+1)%3==st.size()) count++;
        }
        return count;
    }
};

class Solution {
public:
    string trimTrailingVowels(string s) {
        int ind=-1;
        int n=s.size();
        for(int i=n-1; i>=0; i--) {
            if (s[i]!='a' && s[i]!='e' && s[i]!='i' && s[i]!='o' && s[i]!='u') {
                ind=i; break;
            }
        }
        string result="";
        for (int i=0; i<=ind; i++) result+=s[i];
        return result;
    }
};

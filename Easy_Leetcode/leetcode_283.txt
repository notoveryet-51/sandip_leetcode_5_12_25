class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int n=nums.size();
        vector<int>vec;
        for (int i=0; i<n; i++) {
            if (nums[i]!=0) vec.push_back(nums[i]);
        }
        int zero=n-vec.size();
        for (int i=1;i<=zero; i++)  vec.push_back(0);
        nums=vec;
    }
};

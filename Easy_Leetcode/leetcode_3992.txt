class Solution {
public:
    string rearrangeString(string s, char x, char y) {
        string str="";
        int n=s.size(); 
        for (int i=0; i<n; i++) {
            if (s[i]==x || s[i]==y) continue;
            else str+=s[i];
        }
        for (int i=0; i<n; i++) {
            if (s[i]==y) str=s[i]+str;
            else if (s[i]==x) str+=s[i];
        }
        return str;
    }
};

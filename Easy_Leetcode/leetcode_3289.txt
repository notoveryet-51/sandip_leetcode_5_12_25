class Solution {
public:
    vector<int> getSneakyNumbers(vector<int>& nums) {
        int n=nums.size();
        unordered_map<int,int>mp;
        for (int i=0; i<n; i++) {
            mp[nums[i]]++;
        }
        vector<int>v;
        for (auto it:mp) if (it.second>1) v.push_back(it.first);
        return v;
    }
};

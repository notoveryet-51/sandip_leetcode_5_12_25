class Solution {
public:
    bool validDigit(int n, int x) {
        int copy=n;
        if (n<10) {
            return false;
        }
        int first,dig_count=0;
        int count=0;
        while (n>0) {
            dig_count++;
            int dig=n%10;
            if (dig==x) count=1;
            n/=10;
        }
        int div=(int)pow(10,dig_count-1);
        first=copy/div;
        if (first!=x && count==1 ) return true;
        return false;
    }
};

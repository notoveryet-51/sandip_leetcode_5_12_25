class Solution {
public:
    vector<int> arrayRankTransform(vector<int>& arr) {
        vector<int> rank;
        set<int> st;
        for (auto it: arr)  st.insert(it);
        int count=0;
        unordered_map<int, int> mp;
        for (auto it: st) {
            mp.insert({it, ++count});
        }
        for (auto it: arr) {
            rank.push_back(mp[it]);
        }
        return rank;
    }
};

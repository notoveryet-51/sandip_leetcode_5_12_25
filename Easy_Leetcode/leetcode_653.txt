/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void inorderTraverse(TreeNode *root,unordered_map<int,int> &mpp, bool &check, int k) {
        if (root==NULL) return;
        inorderTraverse(root->left, mpp, check, k);
        if (mpp.find(k-root->val)!=mpp.end()) {
            check=true;
        }
        mpp.insert({root->val, 1});
        inorderTraverse(root->right, mpp, check, k);
    }

    bool findTarget(TreeNode* root, int k) {
        bool check=false;
        unordered_map<int,int> mpp;
        inorderTraverse(root, mpp, check, k);
        return check;
    }
};

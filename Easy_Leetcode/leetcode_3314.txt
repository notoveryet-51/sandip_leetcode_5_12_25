class Solution {
public:
    vector<int> minBitwiseArray(vector<int>& nums) {
        vector<int>vec;
        for (int i=0; i<nums.size(); i++) {
            for (int j=0; j<=10000; j++) {
                int check=(j|(j+1));
                if (check==nums[i]) {
                    vec.push_back(j);   break;
                }
            }
            if (vec.size()!=i+1)    vec.push_back(-1);
        }
        return vec;
    }
};

class Solution {
public:
    vector<int> recoverOrder(vector<int>& order, vector<int>& friends) {
        int n1=order.size(), n2=friends.size();
        vector<int>vec;
        for (int i=0; i<n1; i++)    {
            for (int j=0; j<n2; j++)    {
                if (order[i]==friends[j])   vec.push_back(order[i]);
            }
        }
        return vec;
    }
};

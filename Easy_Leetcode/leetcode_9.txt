#include <stdbool.h>

bool isPalindrome(int x) {
    long long n, count = 0, a = 1, b, c, d;
    n = x;   
    b = x;   

    if (n < 0)
        return false;
    else if (n < 10)
        return true;

    
    while (n != 0) {
        n = n / 10;
        count++;
    }

    
    for (int i = 1; i < count; i++) {
        a *= 10;
    }

    while (b != 0) {
        c = b / a;
        d = b % 10;  

        if (c != d)
            return false;

        
        b = (b % a) / 10;
        a /= 100;    
    }

    return true;
}

class Solution {
public:
    int absDifference(vector<int>& nums, int k) {
        int n=nums.size();
        sort (nums.begin(),nums.end());
        int first_sum=0,last_sum=0;
        for (int i=0; i<k; i++) {
            first_sum+=nums[i];
            last_sum+=nums[n-1-i];
        }
        return abs(first_sum-last_sum);
    }
};

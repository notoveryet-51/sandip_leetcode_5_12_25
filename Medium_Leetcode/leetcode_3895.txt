class Solution {
public:
    int countDigitOccurrences(vector<int>& nums, int digit) {
        int count=0; int n=nums.size();
        for (int i=0; i<n; i++) {
            int copy=nums[i];
            while (copy>0) {
                int dig=copy%10;
                if (dig==digit) count++;
                copy/=10;
            }
        }
        return count;
    }
};

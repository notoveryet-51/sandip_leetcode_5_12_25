class Solution {
public:
    int smallestBalancedIndex(vector<int>& nums) {
        int n=nums.size();  long long sum=0,  prd=1;
        for(int i=0; i<n; i++) {
            sum+=nums[i];
        }
        for (int i=n-1; i>=0; i--)  {
            sum-=nums[i];
            if (prd==sum)   return i;
            if (prd>sum)    break;
            if (prd>(LLONG_MAX/nums[i])) break;
            prd*=nums[i];
        }
        return -1;
    }
};

class Solution {
public:
    int jump(vector<int>& nums) {
        int n=nums.size(); int track=0;
        int count=0;
        int currentMax=0;
        if (n == 1) return 0;
        for (int i=0; i<n-1; i++) {
            track=max(track, i+nums[i]);
            if (i==currentMax) {
                count++;
                currentMax=track;
            }
        }
        return count;
    }
};

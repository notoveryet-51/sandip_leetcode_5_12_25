/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void determineDeleteNode(TreeNode* &node) {
        // Case 1: Leaf node or Case 2a: Only right child
        if (node->left == nullptr) {
            TreeNode *temp = node;
            node = node->right;
            delete temp;
        }
        // Case 2b: Only left child
        else if (node->right == nullptr) {
            TreeNode *temp = node;
            node = node->left;
            delete temp;
        }
        // Case 3: Node has two children
        else {
            // Find the inorder predecessor (largest value in the left subtree)
            TreeNode *predecessor = node->left;
            while (predecessor->right != nullptr) {
                predecessor = predecessor->right;
            }
            node->val = predecessor->val;
            deleteFromSubtree(node->left, predecessor->val);
        }
    }

    void deleteFromSubtree(TreeNode* &root, int key) {
        if (root == nullptr) return;
        
        if (root->val > key) {
            deleteFromSubtree(root->left, key);
        } else if (root->val < key) {
            deleteFromSubtree(root->right, key);
        } else {
            determineDeleteNode(root);
        }
    }

    TreeNode* deleteNode(TreeNode* root, int key) {
        TreeNode *temp = root;
        TreeNode *prev = nullptr;
        while (temp != nullptr && temp->val != key) {
            prev = temp;
            if (temp->val > key) {
                temp = temp->left;
            } else {
                temp = temp->right;
            }
        }
        if (temp != nullptr) {
            if (prev == nullptr) {
                determineDeleteNode(root);
            } else if (prev->left == temp) {
                determineDeleteNode(prev->left);
            } else {
                determineDeleteNode(prev->right);
            }
        }
        return root;
    }
};

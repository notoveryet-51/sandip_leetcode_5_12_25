class Solution {
public:
    int maxDistance(string moves) {
        int leftRight=0, upDown=0, count=0;
        for (int i=0; i<moves.size(); i++) {
            if (moves[i]=='U') upDown++;
            else if (moves[i]=='D') upDown--;
            else if (moves[i]=='L') leftRight--;
            else if (moves[i]=='R') leftRight++;
            else {
                count++;
            }
        }
        return abs(leftRight)+abs(upDown)+count;
    }
};

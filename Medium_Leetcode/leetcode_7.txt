class Solution {
public:
    int reverse(int x) {
        int temp=x;
        long long sum=0;
        if (x<0 && x>INT_MIN)    x=-x;
        else if (x>0) x=x;
        else return 0;
        while (x>0) {
            sum=sum*10+x%10;
            if (sum>INT_MAX)    return 0;
            x/=10;
        }
        if (temp<0) return -sum;
        return sum;
    }
};

class Solution {
public:
    int scoreDifference(vector<int>& nums) {
        int n=nums.size();
        int sc1=0,sc2=0;
        int flag=0;
        for (int i=0; i<n; i++) {
            if (nums[i]%2!=0) {
                flag^=1;
            }
            if ((i+1)%6==0) {
                flag^=1;
            }
            if (flag==0) sc1+=nums[i];
            else sc2+=nums[i];
        }
        return sc1-sc2;
    }
};

class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        vector<int>v1,v2,result;
        for (auto it:nums) {
            if (it<0)   v1.push_back(it);
            else v2.push_back(it);
        }
        for (int i=0; i<nums.size()/2; i++) {
            result.push_back(v2[i]);
            result.push_back(v1[i]);
        }
        return result;
    }
};

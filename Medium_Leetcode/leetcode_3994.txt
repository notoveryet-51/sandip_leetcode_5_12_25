class Solution {
public:
    int minAdjacentSwaps(vector<int>& nums, int a, int b) {
        int n=nums.size();
        int countMid=0;
        int countGreatB=0;
        int MOD=1e9+7;
        int swapTotal=0;
        for (int i=0; i<n; i++) {
            if (nums[i]>=a && nums[i]<=b) {
                swapTotal=(swapTotal+countGreatB)%MOD;
                countMid++;
            }
            else if (nums[i]<a) {
                swapTotal=(swapTotal+countGreatB+countMid)%MOD;
            }
            else countGreatB++;
        }
        return swapTotal;
    }
};

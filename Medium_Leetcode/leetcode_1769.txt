class Solution {
public:
    vector<int> minOperations(string boxes) {
        vector<int>vec;
        for (int i=0; i<boxes.size(); i++) {
            if(boxes[i]=='1') vec.push_back(i);
        }
        vector<int>result;
        for (int i=0; i<boxes.size(); i++) {
            int temp=0;
            for (int j=0; j<vec.size(); j++) {
                temp+=abs(vec[j]-i);
            }
            result.push_back(temp);
        }
        return result;
    }
};

class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        int subset=1<<nums.size();
        vector<int>vec;
        vector<vector<int>>v;
        for (int i=0; i<subset; i++) {
            vec.clear();
            for (int j=0; j<nums.size(); j++) {
                if (i & (1 << j)) {
                    vec.push_back(nums[j]);
                }
            }
            v.push_back(vec);
        }
        return v;
    }
};

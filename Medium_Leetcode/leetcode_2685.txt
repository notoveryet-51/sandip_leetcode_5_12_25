class Solution {
public:
    void connectedComponents(vector<vector<int>> &adjList, queue<int> &q, vector<int> &visited, int startNode, int &vertexCount, int &edgeCount) {
        q.push(startNode);
        visited[startNode]=1;
        while (!q.empty())  {
            auto a=q.front();
            q.pop();
            vertexCount++; // Count this vertex
            edgeCount += adjList[a].size(); // Sum the degrees of all vertices in this component
            for (auto it: adjList[a]) {
                if (!visited[it]) {
                    q.push(it);
                    visited[it]=1;
                }
            }
        }
    }
    int countCompleteComponents(int n, vector<vector<int>>& edges) {
        vector<vector<int>>adjList(n);
        // adjacency list
        for (int i=0; i<edges.size(); i++) {
            adjList[edges[i][0]].push_back(edges[i][1]);
            adjList[edges[i][1]].push_back(edges[i][0]);
        }
        queue<int> q;
        vector<int> visited(n, 0);
        int count=0;
        for (int i=0; i<n; i++) {
            if (!visited[i]) {
                int vertexCount = 0; 
                int edgeCount = 0; 
                connectedComponents(adjList, q, visited, i, vertexCount, edgeCount);
                
                // Each undirected edge is counted twice (once from each endpoint)
                if (edgeCount == vertexCount * (vertexCount - 1)) {
                    count++;
                }
            }
        }
        return count;
    }
};

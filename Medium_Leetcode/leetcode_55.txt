class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n=nums.size(); int track=0;
        int i=0;
        while (i <= track) {
            track = max(track, i + nums[i]);
            if (track >= n - 1) return true;
            i++; 
        }
        if (track>=n-1) return true;
        else return false;
    }
};

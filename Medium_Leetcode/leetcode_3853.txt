class Solution {
public:
    string mergeCharacters(string s, int k) {
        int n=s.size(); string ret="";
        for (int i=0; i<n; ++i) {
            for (int j=i+1; j<n; j++) {
                if (s[i]==s[j] && (j-i)<=k) {
                    s.erase(s.begin()+j); i=-1; n--; break;
                }
            }
        }
        return s;
    }
};

class Solution {
public:
    void helperFunction(vector<string> &parenthesis, int open, int close, string str) {
        if (open==0 && close==0) {
            parenthesis.push_back(str);
            return;
        }
        if (open>0) {
            helperFunction(parenthesis, open-1, close, str+"(");
        }
        if (close>open) {
            helperFunction(parenthesis, open, close-1, str+")");
        }
    }
    vector<string> generateParenthesis(int n) {
        int open=n, close=n;
        vector<string> parenthesis;
        string str;
        helperFunction(parenthesis, open, close, str);
        return parenthesis;
    }
};

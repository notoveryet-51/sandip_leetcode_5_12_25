class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        vector <int> vec;
        int n=nums.size();
        if (k>n)    k=k%n;
        int index=n-k;
        for (int i=index; i<n; i++) vec.push_back(nums[i]);
        for (int i=0; i<index; i++) vec.push_back(nums[i]);
        for (int i=0; i<n; i++) nums[i]=vec[i];
        return;
    }
};

class Solution {
public:
    bool uniformArray(vector<int>& nums1) {
        int n=nums1.size();
        int flag=0;
        // all even check
        for (int i=0; i<n; i++) if (nums1[i]%2!=0) flag=1;
        if (flag==0) return true;
        // mixed check
        int mini=nums1[0];
        for (int i=0; i<n; i++) {
            mini=min(nums1[i],mini);
        }
        if (mini%2==0) return false;
        else return true;
    }
};

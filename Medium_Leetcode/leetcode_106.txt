/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* helperFunction(vector<int> &postorder, int postStart, int postEnd, vector<int> &inorder, int inStart, int inEnd, unordered_map<int, int> &inMap)  {
        if (postStart< postEnd || inStart> inEnd) return NULL;
        TreeNode *root= new TreeNode(postorder[postStart]);
        int inRoot=inMap[root->val]; // direct fetching the index through map
        int numsLeft= inRoot-inStart;   //number of elements in the left subtree
        // build leftSubtree
        root->left= helperFunction(postorder, postEnd+numsLeft-1, postEnd ,inorder, inStart, inRoot-1, inMap);
        // build rightSubtree
        root->right= helperFunction(postorder, postStart-1, postEnd+numsLeft, inorder, inRoot+1, inEnd, inMap);
        return root;
    }

    TreeNode* buildTree(vector<int>& inorder, vector<int>& postorder) {
        unordered_map<int, int> inMap;
        // putting the values in map to reference instantly with values
        for (int i=0; i<inorder.size(); i++) inMap[inorder[i]]=i; // <value, index>
        return helperFunction(postorder, postorder.size()-1, 0, inorder, 0, inorder.size()-1, inMap);
    }
};

class Solution {
public:
    vector<long long> mergeAdjacent(vector<int>& nums) {
        vector<long long>v;
        for (int i=0; i<nums.size(); i++)   {
            if (v.empty() || v.back()!=nums[i]) v.push_back(nums[i]);
            else v[v.size()-1]=2*v[v.size()-1];
            while (v.size()>=2 && v[v.size()-1]==v[v.size()-2])   {
                v[v.size()-2]=2*v[v.size()-2];
                v.pop_back();
            }
        }
        return v;
    }
};

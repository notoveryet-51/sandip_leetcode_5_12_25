class Solution {
public:
    vector<int> nextGreaterElements(vector<int>& arr) {
        int n=arr.size();
        vector<int>nums(n);
        stack<int>s;
        for (int i=2*n-1; i>=0; i--)  {
            while (!s.empty() && s.top() <= arr[i%n])
                s.pop();
            if (i<n)    {
                if (s.empty())  nums[i] = -1;
                else    nums[i] = s.top();
            }
            s.push(arr[i%n]);
        }
        return nums;
    }
};

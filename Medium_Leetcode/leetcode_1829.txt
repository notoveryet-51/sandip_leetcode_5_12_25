class Solution {
public:
    vector<int> getMaximumXor(vector<int>& nums, int maxi) {
        int n=nums.size();
        int target=(1<<maxi)-1;
        vector<int>result;
        int x=0;
        for (int i=0; i<n; i++) {
            x^=nums[i];
            result.push_back(x^target);
        }
        reverse(result.begin(),result.end());
        return result;
    }
};
